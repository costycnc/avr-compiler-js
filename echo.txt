.org 0x0000
    rjmp RESET

.org 0x60
RESET:
sbi 4,5;led onboard PB5 13
sbi 4,0;triger PB0  8
cbi 4,1;echo    PB1  9

MAIN_LOOP:
    sbi 5, 0    ;pulse triger
    loop:
    dec r20
    brne loop
    cbi 5, 0   

ldi r28,25 ;wait echo long
    loop1:
    dec r27
    brne loop1
    dec r28
    brne loop1

    cbi 5,5 
    sbic 3,1     ; check echo (led high/low)
    sbi 5,5

ldi r22,10 ;wait first to send new pulse
sus:
dec r20
brne sus
dec r21
brne sus
dec r22
brne sus

rjmp MAIN_LOOP
